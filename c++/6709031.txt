ll n;

int main(){
  cin >> n;
  cout << n << ":";
  while(true){
    bool flag = false;
    for(ll i = 2; i*i <= n; i++){
      if(n%i == 0){
        cout << " " << i;
        n /= i;
        flag = true;
        break;
      }
    }
    if(flag) continue;
    break;
  }
  cout << " " << n << endl;
}
