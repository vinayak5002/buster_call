#include <iostream>
using namespace std;
int main()
{
    int n, x;
    int ans = 0;
    while (cin >> n >> x)
    {
        ans = 0;
        if (n == 0)
            break;
        for (int i = 1; i <= n; i++)
        {
            for (int j = i + 1; j <= n; j++)
            {
                for (int k = j + 1; k <= n; k++)
                {
                    if (i + j + k == x)
                        ans++;
                }
            }
        }
        cout << ans << endl;
    }

    return 0;
}
