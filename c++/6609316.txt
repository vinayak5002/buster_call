#include<iostream>
using namespace std;
int main(){
    int n, x;
    cin >> n >> x;
    while(n!=0||n!=0){
        int i, j, k, l=0;
        for(i=0;i<n;i++){
            for(j=0;j<n;j++){
                for(k=0;k<n;k++){
                    if(i+j+k==x)l++:
                }
            }
        }cout << l << endl;
        cin >> n >> x;
    }
}
