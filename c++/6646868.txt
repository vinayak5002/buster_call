#include <bits/stdc++.h>
#include <stdio.h>
#include <stdlib.h>

using namespace std;
typedef long long ll;

#define N 51
#define M 3

int main()
{
    int i, j, k;

    int n, x, count = 0;
    while(1) {
        scanf("%d %d", &n, &x);
        if (n == 0 && x == 0) {
            break;
        }

        for (i = 1; i <= n; i++) {
            for (j = i + 1; j <= n; j++) {
                for (k = j + 1; k <= n; k++) {
                    if (i + j + k == x) {
                        count++;
                    }
                }
            }
        }
    }

    printf("%d\n", count);

    return 0;
}
