#include<iostream>
using namespace std;

int main(){
    int n, x;
    cin >> n;
    for(int i = 1; i <= n; i++){
        while(1){
            x = i;
            if(x % 3 == 0){
                cout << " " << i;
                break;
            }
            while(x){
                if(x % 10 == 3){
                    cout << " " << i;
                    break;
                }
                x /= 10;
            }
        }
    }
    cout << endl;
    return 0;
}
