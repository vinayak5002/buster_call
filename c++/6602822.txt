#include<stdio.h>
int main(void){
    int a[4];
    int i; //foræ–‡
    for(i=0; i<4; i++){
        scanf("%d",&a[i]);
    }
    
    int j,k,memo=0;
    for(j=0; j<4; j++){
        for(k=0; k<4; k++){
            if(a[j] == a[k]){
                memo++;
            }
        }
    }
    
    int ret;
    if(memo==6){
        ret = 1;
    }else{
        ret = 0;
    }
    
    printf("%d",ret);
    
    return 0;
}
