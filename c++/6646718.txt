int main() {
	while (true)
	{
		int n, m;
		cin >> n >> m;
		if(n == 0 and m == 0) return 0;
		vector v(m, vector<i64>{});
		rep(i, n) {
			i64 nd, nv;
			cin >> nd >> nv;
			nd--;
			v[nd].push_back(nv);
		}
		for(auto & nv : v) sort(rall(nv));
		i64 ans = 0;
		for(auto & nv : v) {
			// cout << nv << endl;
			ans += nv[0];
		}
		cout << ans << endl;
	}
	

	return 0;
}
