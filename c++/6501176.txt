#include<iostream>
using namespace std;
int main()
{
    int n;
    cin>>n;
    int arr[n];
    for(int i=0;i<n;i++)
       cin>>arr[i];
    for(int i=1;i<n;i++)
    {
        for(int i=0;i<n;i++)
            cout<<arr[i]<<" ";
        cout<<"\n";
        int j=i-1;
        int x=arr[i];
        while(j>=0 && arr[j]>x)
        {
            arr[j+1]=arr[j];
            j--;
        }
        arr[j+1]=x;
        
    }
    for(int i=0;i<n;i++)
        cout<<arr[i]<<" ";
    cout<<"\n";
}
