#include <stdio.h>

main(void){
	int n,m,i,j;
	int a[255][255] = {{0}};
	int b[255] = {0};
	int c[255] = {0};
	
	
	scanf("%d %d",&n,&m);
	
	
	for(i=0; i<n; i++){
		for(j=0;j<m;j++){
			scanf("%d",&a[i][j]);
		}
	}
	
	for(i=0; i<m; i++){
		scanf("%d",&b[i]);
	}
	
	for(i=0; i<n; i++){
		for(j=0; j<m; j++){
			c[i] += a[i][j]*b[j];
		}
		printf("%d\n",c[i]);
	}
	
}
