int n;
vector<pair<int, int> > seen;
vector<vector<int> > G;
int cnt = 0;

void dfs(int v){
  cnt++;
  seen[v].first = cnt;
  for(auto g: G[v]){
    if(seen[g].first == -1){
      dfs(g);
    }
  }
  // 帰りも++するのが重要
  cnt++;
  seen[v].second = cnt;
}
