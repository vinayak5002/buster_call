#include <iostream>
using namespace std;
 
int main(){
  int h,w;
  cin >> h >> w;
  while(!(h==0 & w==0)){
      for(int i = 0; i < h; i++){
          if(i==0 || i < h-1){
              for(int i = 0; i < w; i++){
                  cout << "#";
          }else{
              for(int i = 0; i < w; i++){
                  if(i==0 || i < w-1){
                        cout << "#";
                  }else{
                        cout << ".";
                  }
              }
          }
          cout << "\n";
       }
      cout << "\n";
      cin >> h >> w;
  }
  return 0;
}
