#include <stdio.h>
int main()
{
    int n, x, i, j, k, a=0, b=0;
    
    while(1){
     scanf("%d %d", &n, &x);
      if(n==0 && x==0)
          break;
    }
     for(i=1; i<n; i++){
         for(j=1; j<n; j++){
             for(k=1; k<n; k++){
                 if(i==j || j==k)
                    continue;
                    
                    a = i+j+k;
                     if(a==x)
                      b++;
                 
             }
         }
     printf("%d\n", b);
     }
     
     
     return 0;
}
