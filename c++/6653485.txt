#include<stdio.h>
#include<iostream>
using namespace std;

int main(){
    int H, W;
    
    while (1) {
      	if (H == 0 && W == 0) {
          break;
        }
      
        cin >> H >> W;
        for (int i = 0; i < H; i++) {
            for (int i = 0; i < W; i++) {
                cout << "#";    
            }
            cout << endl;
        }
        cout << endl;
    }
}
