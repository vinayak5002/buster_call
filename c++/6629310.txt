#include<iostream>
#include<cctype>
using namespace
int main(){
    char ch;
    if ( ch == '\n' )break;
    while(1){
        scanf("%c", &ch);
        if ('a' <= ch && ch <= 'z');
        ch = ch + 32
        printf("%c = %d\n", ch, ch);
        else
        ch = ch - 32
        printf("%c = %d\n", ch, ch);
    }
    return 0
}
