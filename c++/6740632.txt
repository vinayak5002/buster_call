void solve(){
	int N;
	cin>>N;
	string S;
	cin>>S;
	vector<int> dp(8);
	dp[1]=1;
	rep(i,N){
		vector<int> n_dp(8);
		int tmp=1;
		if(S[i]=='O') tmp=2;
		if(S[i]=='I') tmp=4;
		rep(k,8) rep(l,8){
			if((l&tmp)!=0&&(k&l)!=0){
				n_dp[l]=(n_dp[l]+dp[k])%mod;
			}
		}
		swap(n_dp,dp);
	}
	int ans=0;
	rep(i,8) ans=(ans+dp[i])%mod;
	cout<<ans<<"\n";
}
