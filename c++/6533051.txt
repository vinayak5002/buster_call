#include<stdio.h>
int main(void){
    int  n,min=0,max=0,sam=0,m;
    long long i[n];
    scanf("%d",&n);
    for(m=0;m<n;m++){
        scanf("%lld",&i[m]);
        sam=sam+i[m];
        if(m>0&&i[m]<i[m-1]){
            min=i[m];
        }
        if(m>0&&i[m]>i[m-1]){
            max=i[m];
        }
    }
    printf("%d %d %dÂ¥n",min,max,sam);
    return 0;
}
