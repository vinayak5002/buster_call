using Graph = vector<vector<int>>;
class LCA {
    static constexpr int NONE = -1;
    vector<int> dist;
    vector<vector<int>> parents;
public:
    LCA(const Graph &G) {
        int V = G.size();
        int K = 1;
        while (1<<K < V) ++K;

        dist.assign(V, 0);
        parents.assign(K, vector(V, NONE));

        dfs(G, 0);

        FOR(k, 0, K-1) {
            FOR(v, 0, V) {
                if (parents[k][v] == NONE) {
                    parents[k+1][v] = NONE;
                } else {
                    parents[k+1][v] = parents[k][parents[k][v]];
                }
            }
        }
    }

    void dfs(const Graph &G, int v, int d=0, int p=NONE) {
        dist[v] = d;
        parents[0][v] = p;
        for (auto nv : G[v]) {
            if (nv == p) continue;
            dfs(G, nv, d+1, v);
        }
    }

    int query(int u, int v) {
        if (dist[u] < dist[v]) swap(u, v);
        int K = parents.size();

        int diff = dist[u] - dist[v];
        FOR(k, 0, K) {
            if (diff>>k & 1) u = parents[k][u];
        }

        if (u == v) return u;

        FORR(k, K-1, 0) {
            if (dist[u] == dist[v]) continue;
            u = parents[k][u];
            v = parents[k][v];
        }
        return parents[0][u];
    }
};

void _main() {
    int n;
    cin >> n;

    Graph G(n);
    REP(i, n) {
        int c, k; cin >> k;
        REP(j, k) {
            cin >> c;
            G[i].push_back(c);
        }
    }

    LCA lca(G);
    int q, u, v; cin >> q;
    REP(i, q) {
        cin >> u >> v;
        cout << lca.query(u, v) << endl;
    }
}

int main() {
    _main();
    return 0;
