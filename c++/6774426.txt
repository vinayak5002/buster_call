#include<stdio.h>

int main(){
    int H,W,a,b;
    while(1){
    scanf("%d%d", &H, &W);
    if(H==0 && W==0){
        break;
    }
    for(a=0;a<H;a++){
        for(b=0;b<W;b++){
            if(b==0 || b==W-1 || a==0 || a==H-1){
                printf("#");
            }else{
                printf(".");
            }
        }
        printf("\n");
    }
    printf("\n");
    }
    return 0;
}
