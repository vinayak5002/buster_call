#include<iostream>
#include<string>
using namespace std;

int main(){
    string str1, str2;
    int n, point1=0, point2=0;
    cin>>n;
    for(int i=0; i<n; i++){
        cin>>str1>>str2;
        if(str1==str2){
            point1++; point2++;
        }else if(str1 > str2{
            point1+=3;
        }else{
            point2+=3;
        }
    }
}
