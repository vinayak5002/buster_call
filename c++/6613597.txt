#include <stdio.h>
int main(){
    int h,w,i,j;
    while(1){
        scanf("%d %d ",&h,&w);
        if(h == 0 && w == 0)break;
        for(j=0;j<w;j++){printf("#");}
        for(i=0;i<h-1;i++){
                printf("#");
            for(j=1;j<w-1;j++){
                printf(".");
            }                
            printf("#");
            }
            for(j=0;j<w;j++){printf("#");
            printf("\n");}
            printf("\n");
    }
    return 0;
}
