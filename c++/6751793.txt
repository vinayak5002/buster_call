template<class Tp>
bool zero (const Tp &x) {
    return x == 0;
}

template<class Tp, class... Args>
bool zero (const Tp &x, const Args& ...args) {
    return zero(x) and zero(args...);
}
int main(){
  while(true){
    double r0,w0,c,r;cin >> r0 >> w0 >> c >> r;
    if(zero(r0,w0,c,r)) break;
    int cnt = 0;
    while(true){
      double y = (r0 + cnt * r - c * w0) / c;
      if(y >= 0)break;
      cnt++;
    }
    cout << cnt << endl;
  }
