#include<stdio.h>
int main(){
    int N,X,i,j,k,r;
    for(;scanf("%d%d",&N,&X),X+N;printf("%d\n",r)){
    	for(i=1,r=0;i<=N;i++){
            for(j=i+1;j<=N;j++){
                k = X-i-j;
            	if(j<k&&k<=N){
                    r++;
            	}
            }
    	}
    }
    return 0;
}
