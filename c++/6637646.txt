#include <stdio.h>

int max(int a, int b)
{
    return a > b ? a : b;
}

int main() 
{
    int n, bag;//n为物体个数，bag为背包容量
    //printf("\n请输入物体个数和背包容量：\n");

	int v[100];//每个物体价值
    //printf("请依次输入每个物体的重量和价值：\n");
    for(int i = 1; i <= n; i++)
    {
        scanf("%d %d", &w[i], &v[i]);
    }
	int maxvalue[100][100] = {0};//动态规划表

				maxvalue[i][wight] = max(maxvalue[i-1][wight], maxvalue[i-1][wight-w[i]] + v[i]);
		}
	}
