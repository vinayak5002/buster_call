#include <stdio.h>

main(void){
	int n,x,i,j,k;
	int ans = 0;
	
	while(1){
		scanf("%d %d", &n,&x);
		if(n==0 && x==0){
			break;
		}
		
		for(i=1;i<=n;i++){
			for(j=i+1;j<=n;j++){
				for(k=j+1;k<=n;k++){
					if((i+j+k)==x){
						ans++;
					}
				}
			}
		}
		printf("%d\n",ans);	
		ans=0;
	}
}
