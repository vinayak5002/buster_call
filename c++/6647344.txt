#include <iostream>
using namespace std;



int main()
{
	int n, x;
	

	while (1) {

		cin >> n >> x;

		if (n == 0 && x == 0) {
			break;
		}

		int count = 0;

		for (int i = 1; i <= n; i++) {
			for (int j = 1; j <= n; j++) {
				for (int k = 1; k <= n; k++) {
					if (i != j && j != k && k != i) {
						if (i + j + k == x) {
							count += 1;
						}
					}
				}
			}
		}

		cout << count << endl;
	}

	



	return 0;
}
