#include <bits/stdc++.h>
using namespace std;
int main(){
    while(1){
        int n,x;
        cin>>n>>x;
        if(n==0&&x==0){
            break;}
    int ans;
    for(int i=1;i<=n;i++){
        for(int j=i+1;j<-n;j++){
            for(int k=i+j+1;k<=n;k++){
                if(i+j+k==x){
                    ans++;
                }
                }
            }
        }
       
    }
       cout<<ans<<endl;  
    }
