#include<iostream>
#include<algorithm>
using namespace std;
int main(){
  int n,A[100],b=0;
  cin>>n;
  for(int i=0;i<n;i++)cin>>A[i];
  for(int i=0;i<n-1;i++){
    for(int j=n-1;j>i;j--){
      if(A[j]<A[j-1]){
    swap(A[j],A[j-1]);
    b++;
      }
    }
  }
  for(int i=0;i<n;i++){
    if(i)cout<<' ';
    cout<<A[i];
  }
  cout<<endl<<b<<endl;
  return 0;
}
