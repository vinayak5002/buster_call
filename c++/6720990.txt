template <class T>
T pow(T x, i64 n) {
    T res = 1;
    while (n > 0) {
        if (n & 1) res = res * x;
        x = x * x;
        n >>= 1;
    }
    return res;
}

template <class T>
T mod_pow(T x, i64 n, i64 mod) {
    T res = 1;
    while (n > 0) {
        if (n & 1) res = (res * x) % mod;
        x = (x * x) % mod;
        n >>= 1;
    }
    return res;
}

namespace ebi {

void main_() {
    i64 s;
    std::cin >> s;
    i64 now = 1;
    auto f = [&](i64 lower, i64 upper) -> i64 {
        i64 size = 0;
        {
            i64 ret = lower;
            while(ret > 0) {
                ret /= 10;
                size++;
            }
        }
        assert(upper <= lower * 10);
        i64 res = (upper - lower) * size;
        i64 three = (upper - 1)/3 - (lower - 1)/3;
        i64 five = (upper - 1)/5 - (lower - 1)/5;
        i64 fifteen = (upper - 1)/15 - (lower - 1)/15;
        res -= (three + five - fifteen) * size;
        res += three * 4 + five * 4;
        return res;
    };
    while(1) {
        i64 ret = f(now, now * 10);
        if(s <= ret) {
            i64 ok = now, ng = 10*now;
            while(std::abs(ok- ng) > 1) {
                i64 mid = (ok + ng)/2;
                i64 x = f(now, mid);
                if(x < s) {
                    ok = mid;
                }
                else {
                    ng = mid;
                }
            }
            s -= f(now, ok);
            assert(s > 0);
            std::string ans = "";
            now = ok;
            while(ans.size() < 50) {
                if(now % 3 == 0) {
                    ans += "Fizz";
                }
                if(now % 5 == 0) {
                    ans += "Buzz";
                }
                if(now % 3 != 0 && now % 5 != 0) {
                    ans += std::to_string(now);
                }
                now++;
            }
            std::cout << ans.substr(s-1, 20) << '\n';
            break;
        }
        s -= ret;
        now *= 10;
    }
}

}  // namespace ebi

int main() {
    std::cout << std::fixed << std::setprecision(15);
    std::cin.tie(nullptr);
    std::ios::sync_with_stdio(false);
    int t = 1;
    // std::cin >> t;
    while (t--) {
        ebi::main_();
    }
    return 0;
}
