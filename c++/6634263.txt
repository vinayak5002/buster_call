ll power(ll x, ll n) {
    ll res = 1;
    while (n>0) {
        if (n & 1) {
            res *= x;
        }
        x *= x;
        n >>= 1;
    }
    return res;
}

    ll m, n;
    cin >> m >> n;
    cout << power(m, n) << endl;

}
