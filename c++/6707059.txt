#include <cstdio>

int main()
{
    int H = 1, W = 1;
    while (1)
    {
        scanf("%d %d", &H, &W);
        if ((H == 0) && (W == 0))
            break;
        for (size_t i = 0; i < H; i++)
        {
            for (size_t j = 0; j < W; j++)
            {
                if((i == 0) || (i == H-1) || (j == 0) || (j == W-1))printf("#");
                else printf(".");
            }
            printf("\n");
        }
        printf("\n");
    }
    return 0;
}
