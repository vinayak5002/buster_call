import java.util.Scanner;

public class Main {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int S = sc.nextInt();

		int h, m, s;

		h = S / 3600;
		S = S - h * 3600;
		m = S / 60;
		S = S - m * 60;
		s = S;

		System.out.println(h + ":" + m + ":" + s);

	}

}
