
#include <stdio.h>

int table[105][105];

int main(void) {
	int i, j, n, q;
	int id, t;
	char buff[1024];
	
	scanf("%d\n", &n);
	for(i=0;i<=n;i++) {
		for(j=0;j<=n;j++) {
			table[i][j] = 0;
		}
	}
	
	for(i=0;i<n;i++) {
		scanf("%d %d", &id, &q);
		for(j=0;j<q;j++) {
			scanf("%d", &t);
			table[id][t] = 1;
		}
		gets(buff);//標準入力からの文字列入力
	}
	
	for(i=1;i<=n;i++) {
		for(j=1;j<=n;j++) {
			printf("%d", table[i][j]);
			if(j<n) {
				printf(" ");
			} else {
				printf("\n");
			}
		}
	}
	
	return 0;
}
