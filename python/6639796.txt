import java.util.Scanner; 

public class Main{
    public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        int S = sc.nextInt();
        int h = S/3600;
        int m = h/60;
        int s = m/60;
        
        System.out.println(h+":"+m+":"+s);
        
    }
}
