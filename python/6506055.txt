import java.util.Scanner;

public class Main{
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		
		int n = sc.nextInt();
		int[] a = new int[n];
		int nn = n;

		for(int i=0;i<n;i++) {
			a[i] = sc.nextInt();
		}
		
		
		
		for(int i=0;i<n;i++) {
			System.out.print(a[nn-1]);
			if(i!=n-1) {
				System.out.print(" ");
			}
			nn--;
		}
		
	}

}
