public class Main{
    void insertionSort(A, N){
        for (int i=0; i<N-10; i++){
            v = A[i];
            int j = i-1
            while(j>=0 && A[j] > v){
                A[j+1] = A[j];
                j--;
                A[j+1] = v;
            }
            
        }
        
        
        
        
        
    
    }

    }
}
