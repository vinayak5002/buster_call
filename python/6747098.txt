// PRoblem AE - Min, Max, Sum.cpp : This file contains the 'main' function. Program execution begins and ends there.
//

#include <iostream>
#include <cmath>
#include <climits>
using namespace std;

int main()
{
    int n;
    cin >> n;
    int min_num = INT_MAX, max_num = INT_MIN;
    long long sum = 0;
    for (int i = 0; i < n, i++)
    {
        int x; cin >> x;
        min_num = min(min_num, x);
        max_num = max(max_num, x);
        sum += x;

    }
    cout << min_num << " " << max_num << " " << sum << endl;
}
