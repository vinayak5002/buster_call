import java.util.Scanner;
 
public class Main{
    public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int[] a = new int[n];
        for ( int i = n; 0 < i; i++ ){
                System.out.print(a[i]);
            }
        System.out.println();
    }
}
