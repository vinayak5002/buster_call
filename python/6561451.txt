using System;

{

    {

        for(int i = 0; i < cou; i++){
            var target = alp(i);
            result_num = 0;
            for (int j = 0; j <str_sample.length ; j++){
                if(string.Compare(target, str_sample.Substring(j, 1), true)){
                    result_num += 1;
                }
            }
            array[i] = target.toString() + " ï¼š " + result_num.toString(); 
        }
        
        for(int i = 0;i<array.length;i++){
