
#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;

int main()
{
    int n; cin >> n;
    vector<int> v1(n);
    for (int i = 0; i < n; i++) cin >> v1[i];
    
    int m; cin >> m;
    vector<int> v2(m);
    for (int i = 0; i < m; i++) cin >> v2[i];
    
    vector<int> res(m + n);
    auto it = set_union(begin(v1), end(v1), begin(v2), end(v2), begin(res));
    res.resize(it - res.begin());
    for (auto p : res) cout << p << "\n";
}
