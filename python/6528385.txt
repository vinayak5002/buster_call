  int n;
  cin >> n;
  vector<ll> vs;
  rep(i, n) {
    string op;
    ll value;
    cin >> op;
    if (op == "insert") {
      cin >> value;
      vs.push_back(value);
    } else if (op == "delete") {
      cin >> value;
      auto it = find(all(vs), value);
      if (*it == value) {
        vs.erase(it);
      }
    } else if (op == "deleteFirst") {
      vs.pop_back();
    } else if (op == "deleteLast") {
      vs.erase(vs.begin());
    }
  }
  reverse(all(vs));
  dump(vs);
}
