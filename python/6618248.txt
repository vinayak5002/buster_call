import java.util Scanner;

class Main {
    public static void main(String[] args) {
        Scanner scan = new Scanner(System.in);
        
        int second = scan.nextInt();
        
        int hour = second / 3600;
        int min = (second % 3600) / 60;
        int sec = min % 60;
        
        System.out.println(hour + ":" + min + ":" + sec);
    }
}
