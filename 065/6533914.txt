


#include<stdio.h>
int main(){
    int i,a,b;
    char op;
    
    for (i=1;i>0;i++){
        scanf("%d %c %d",&a ,&op ,&b);
        
        if(op=='+'){
            printf("%d\n",a+b);
        }
        else if(op== '-'){
            printf("%d\n",a-b);
        }
        else if(op=='*'){
            printf("%d\n",a*b);
        }
        else if(op=='/'){
            printf("%d\n",a/b);
        }
        
        else {
            break;
        }
        
        return 0;
        
        
        
        
    }
    
    
    
    
    
}
