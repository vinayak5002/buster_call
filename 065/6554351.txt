import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        while (true) {
            int x = sc.nextInt();
            int y = sc.nextInt();

            if (x > y) {
                int z = x;
                x = y;
                y = z;
            }

            if (x == 0 && y == 0)
                break;

            System.out.println(x + " " + y);
        }
    }
}
