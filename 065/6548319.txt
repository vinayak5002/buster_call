#チェスボードの描画
"""
以下のような、たてH cm よこ W cm のチェック柄の長方形を描くプログラムを作成して下さい。

#.#.#.#.#.
.#.#.#.#.#
#.#.#.#.#.
.#.#.#.#.#
#.#.#.#.#.
.#.#.#.#.#

上図は、たて 6 cm よこ 10 cm の長方形を表しています。

長方形の左上が "#" となるように描いて下さい。

"""

while True:
  H,W = map(int,input().split())
  i = 1 

  if H ==0 and W == 0:
    break
  
  for i in range(H):
    if i %2 == 0: #　Hが偶数かどうかの判定
      print("#."*(W//2)) # Hが偶数の場合
      i = i+1
    else:
      print(".#"*(W//2)) #　Hが奇数の場合
      i = i+1

  print()
