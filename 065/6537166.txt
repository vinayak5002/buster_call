#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;
int main(){
    int a,b,c;
    cin>>a>>b>>c;
    std::sort<int> array=[a,b,c];
    sort(array.begin,array.end().[](int a, int b)) {
        return a>b;
    }
    cout << array[0] << array[1] << array[2];
    return 0;
}
