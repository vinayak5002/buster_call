#include<stdio.h>
int main(void){
    long long n,a[n],min=0,max=0,sam=0,i;
    scanf("%d",&n);
    for(i=0;i<n;i++){
        scanf("%lld",&a[i]);
        sam=sam+a[i];
        if(i>0&&a[i]<a[i-1]){
            min=a[i];
        }
        if(i>0&&a[i]>a[i-1]){
            max=a[i];
        }
    }
    printf("%d %d %dÂ¥n",min,max,sam);
    return 0;
}
