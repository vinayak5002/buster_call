#include <stdio.h>
int main(void)
{
    int a, b;
    char op;
    
    while(op != ?){
        scanf("%d %c %d", &a, &op, &b);
        if (op == +){
            printf("%d", a + b);
        }
        else if (op == -){
            printf("%d", a - b);
        }
        else if (op == *){
            printf("%d", a * b);
        }
        else if (op == /){
            printf("%d", a / b);
        }
    }
    
    return 0;
}
