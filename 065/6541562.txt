#include <stdio.h>


int main(void) {
    int n, m, l, hyo1[100][100] = {0},hyo2[100][100] = {0}, result[100][100] = {0};
    scanf("%d %d %d", &n, &m, &l);
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            scanf("%d", &hyo1[i][j]);
        }
    }
    for (int i = 0; i < m; i++) {
        for (int j = 0; j < l; j++) {
            scanf("%d", &hyo2[i][j]);
        }
    }
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < l; j++) {
            for (int k = 0; k < m; k++) {
                result[i][j] += hyo1[i][k] * hyo2[k][j];
            }
        }
    }


    for (int i = 0; i < n; i++) {
        for (int j = 0; j < l; j++) {
            printf("%d ", result[i][j]);
        }
        printf("\n");
    }


    return 0;
}
