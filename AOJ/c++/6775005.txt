ll mypow(ll a, ll n)
{
    ll ret = 1;
    if (n == 0)
        return 1;
    if (a == 0)
        return 0;
    rep(i, n)
    {
        if (ret > (ll)(9e18 + 10) / a)
            return -1;
        ret *= a;
    }
    return ret;
}
long long modpow(long long a, long long n, long long mod)
{
    long long res = 1;
    while (n > 0)
    {
        if (n & 1)
            res = res * a % mod;
        a = a * a % mod;
        n >>= 1;
    }
    return res;
}
long long modinv(long long a, long long m)
{
    long long b = m, u = 1, v = 0;
    while (b)
    {
        long long t = a / b;
        a -= t * b;
        swap(a, b);
        u -= t * v;
        swap(u, v);
    }
    u %= m;
    if (u < 0)
        u += m;
    return u;
}

void solve(){
    ll n, m, k, cnt = 0, sum = 0, ans = 0;
    while(1){
        cin>>n>>m;
        if(n==0){
            return;
        }
        vll a(n);
        vll b(n);
        vin(a);
        vin(b);
        vvll c(2,vll(n+2));
        rep(i,n){
            c[0][i+1]=(b[i]+m-a[i])%m;
            c[1][i+1]=(b[i]+m-a[i])%m+m;
        }
        vvll dp(2,vll(n+2,INF));
        dp[0][0]=0;
        rep(i,n+1){
            chmin(dp[0][i+1],dp[0][i]+abs(c[0][i+1]-c[0][i]));
            chmin(dp[0][i+1],dp[1][i]+abs(c[0][i+1]-c[1][i]));
            chmin(dp[1][i+1],dp[0][i]+abs(c[1][i+1]-c[0][i]));
            chmin(dp[1][i+1],dp[1][i]+abs(c[1][i+1]-c[1][i]));
        }
        ans=min(dp[0][n+1],dp[1][n+1]);
        pt(ans/2);
    }
}
