// 繰り返し自乗法 O(log n)
/* pow(x, n)
   = 1 (n == 0)
   = power(xx, n/2) (nが偶数）
   = power(xx, n/2) * x (nが奇数）
   以下は、power(xx, n/2) == 2 * power(x, n/2) を利用
*/
using ull = unsigned long long;
ull myPower(ull x, ull n, ull MOD) {
  ull res = 1;
  if (n > 0) {
    res = myPower(x, n / 2, MOD);
    res = (res * res) % MOD;
    if (n % 2 == 1) res = (res * x) % MOD;
  }
  return res;
}

struct Solver {
  void solve() {
    /* input */
    ll m, n;
    cin >> m >> n;

    int MOD = 1e9 + 7;

    /* solve */
    unsigned long long ans = myPower(m, n, MOD);

    /* output */
    cout << ans << endl;
  }
};
