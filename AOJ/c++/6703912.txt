#include<iostream>
#include<algorithm>
using namespace std;
 
int n,mini;
int A[36],B[36];
char a[36],b[36];
 
string check(){
  for(int i=0;i<n;i++){
    if(a[i]==b[i]&&A[i]==B[i])continue;
    return "Not stable";
  }
  return "Stable";
}
 
int main(){
  cin>>n;
 
  for(int i=0;i<n;i++){
    cin>>a[i]>>A[i];
    b[i]=a[i];
    B[i]=A[i];
  }
 
  for(int i=0;i<n-1;i++){
    for(int j=n-1;j>i;j--){
      if(A[j]<A[j-1]){
    swap(a[j],a[j-1]);
    swap(A[j],A[j-1]);
      }
    }
  }
    for(int i=0;i<n-1;i++){
    mini=i;
    for(int j=i+1;j<n;j++){
      if(B[j]<B[mini])mini=j;
    }
    swap(b[i],b[mini]);   
    swap(B[i],B[mini]);
  }
 
  for(int i=0;i<n;i++){
    if(i)cout<<' ';
    cout<<a[i]<<A[i];
  }
  cout<<endl<<"Stable"<<endl;
  for(int i=0;i<n;i++){
    if(i)cout<<' ';
    cout<<b[i]<<B[i];
  }
  cout<<endl<<check()<<endl;
  return 0;
