#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;
int main(){
    std::sort<int> array[3];
    cin>>array[0]>>array[1]>>array[2];
    sort(array.begin,array.end().[](int a, int b)) {
        return a>b;
    }
    cout << array[0] << array[1] << array[2];
    return 0;
}
