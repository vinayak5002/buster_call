#include <stdio.h>

int partition( int*, int, int );

int main( void ){
  int n, q;
  scanf("%d", &n);

  int A[n];
  for (int i = 0; i < n; i++) { scanf("%d", &A[i]); }

  q = partition(A, 0, n-1);

  for (int i = 0; i < n; i++) {
    if( i != q ) { printf("%d ", A[i]); }
    else {
      printf("[%d] ", A[i]);
   }
  }
  printf("\n");

  return 0;
}

int partition( int *A, int p, int r ){
  int x, i, temp;
  x = A[r];
  i = p-1;
  for (int j = p; j < r; j++) {
    if (A[j] <= x) {
      i = i+1;
      temp = A[i];
      A[i] = A[j];
      A[j] = temp;
    }
  }
  temp = A[i+1];
  A[i+1] = A[r];
  A[r] = temp;
  return i+1;
}
