int main(void){
    int a,b,c,x;
    scanf("%d %d %d",&a,&b,&c);
    if(a>b){
        x=a;
        a=b;
        b=x;
    }
    if(a>c){
        x=a;
        a=c;
        c=x;
        if(b>c){
        x=c;
        c=b;

        }
    }

}
