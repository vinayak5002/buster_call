#include<stdio.h>
int main(void){
    int a[6],i,j=0,t;
    char b[100];
    for(i=0;i<6;i++){
        scanf("%d",&a[i]);
    }
    scanf("%s",b);
    while(1){
        if(b[j]=='\0'){
            break;
        }
        if(b[j]=='N'){
            t=a[0];
            a[0]=a[1];
            a[1]=a[5];
            a[5]=a[4];
            a[4]=t;
        }
        if(b[j]=='E'){
            t=a[0];
            a[0]=a[3];
            a[3]=a[5];
            a[5]=a[2];
            a[2]=t;
        }
        if(b[j]=='S'){
            t=a[0];
            a[0]=a[4];
            a[4]=a[5];
            a[5]=a[1];
            a[1]=t;
        }
        if(b[j]=='W'){
            t=a[0];
            a[0]=a[2];
            a[2]=a[5];
            a[5]=a[3];
            a[3]=t;
        }
        j++;
    }
    printf("%d\n",a[0]);
}
