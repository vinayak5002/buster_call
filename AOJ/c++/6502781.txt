#include<stdio.h>
#include<ctype.h>

void str_toupper(char s[]){
    int i = 0;
    while (s[i]){
        s[i] = toupper[s[i]];
        i++
    }
}

void str_tolower(char s[]){
    int i = 0;
    while (s[i]){
        s[i] = tolower(s[i]);
        i++
    }
}
