#include <stdio.h>
#include <stdlib.h>
#include <math.h>

int main()
{
	int x[2],y[2],w[2],h[2];
	int answer;
	int tm1;
	
	for(tm1=1;tm1<=2;tm1++)
	{
		scanf("%d %d %d %d",&x[tm1-1],&y[tm1-1],&w[tm1-1],&h[tm1-1]);
	}
	
	if((x[0]<=x[1])&&(y[0]<=y[1]))
	{
		if(x[0]+w[0]<x[1]+w[1]&&y[0]+h[0]<y[1]+h[1])
		{
			answer=w[0]*h[0]+w[1]*h[1]-2*(x[0]+w[0]-x[1])*(y[0]+h[0]-y[1]);
		}
		else if(x[0]+w[0]<x[1]+w[1]&&y[0]+h[0]==y[1]+h[1])
		{
			answer=w[0]*h[0]+w[1]*h[1]-2*(x[0]+w[0]-x[1])*(y[0]+h[0]-y[1]);
		}
		else if(x[0]+w[0]==x[1]+w[1]&&y[0]+h[0]<y[1]+h[1])
		{
			answer=w[0]*h[0]+w[1]*h[1]-2*(x[0]+w[0]-x[1])*(y[0]+h[0]-y[1]);
		}
		
		else if(x[0]+w[0]>x[1]+w[1]&&y[0]+h[0]<y[1]+h[1])
		{
			answer=w[0]*h[0]+w[1]*h[1]-2*(w[1])*(y[0]+h[0]-y[1]);
		}
		else if(x[0]+w[0]<x[1]+w[1]&&y[0]+h[0]>y[1]+h[1])
		{
			answer=w[0]*h[0]+w[1]*h[1]-2*(h[1])*(x[0]+w[0]-x[1]);
		}
		
		else
		{
			answer=w[0]*h[0]-w[1]*h[1];
		}		
	}
	
	if((x[1]<x[0])&&(y[1]<y[0]))
	{
		if(x[1]+w[1]<x[0]+w[0]&&y[1]+h[1]<y[0]+h[0])
		{
			answer=w[0]*h[0]+w[1]*h[1]-2*(x[1]+w[1]-x[0])*(y[1]+h[1]-y[0]);
		}
		else if(x[1]+w[1]<x[0]+w[0]&&y[1]+h[1]==y[0]+h[0])
		{
			answer=w[0]*h[0]+w[1]*h[1]-2*(x[1]+w[1]-x[0])*(y[1]+h[1]-y[0]);
		}
		else if(x[1]+w[1]==x[0]+w[0]&&y[1]+h[1]<y[0]+h[0])
		{
			answer=w[0]*h[0]+w[1]*h[1]-2*(x[1]+w[1]-x[0])*(y[1]+h[1]-y[0]);
		}
		
		else if(x[1]+w[1]>x[0]+w[0]&&y[1]+h[1]<y[0]+h[0])
		{
			answer=w[1]*h[1]+w[0]*h[0]-2*(w[0])*(y[1]+h[1]-y[0]);
		}
		else if(x[1]+w[1]<x[0]+w[0]&&y[1]+h[1]>y[0]+h[0])
		{
			answer=w[1]*h[1]+w[0]*h[0]-2*(h[0]*x[1]+w[1]-x[0]);
		}
		
		else
		{
			answer=w[1]*h[1]-w[0]*h[0];
		}		
	}
	
	if((x[1]<x[0])&&(y[1]>=y[0]))
	{
		if(x[1]+w[1]>x[0]+w[0]&&y[1]+h[1]<y[0]+h[0])
		{
			answer=w[0]*h[0]+w[1]*h[1]-2*w[0]*h[1];
		}
		else if(x[1]+w[1]==x[0]+w[0]&&y[1]+h[1]<y[0]+h[0])
		{
			answer=w[0]*h[0]+w[1]*h[1]-2*w[0]*h[1];
		}
		else if(x[1]+w[1]>x[0]+w[0]&&y[1]+h[1]==y[0]+h[0])
		{
			answer=w[0]*h[0]+w[1]*h[1]-2*w[0]*h[1];
		}
		else if(x[1]+w[1]==x[0]+w[0]&&y[1]+h[1]==y[0]+h[0])
		{
			answer=w[0]*h[0]+w[1]*h[1]-2*w[0]*h[1];
		}

		else if(x[1]+w[1]<x[0]+w[0]&&x[1]+w[1]>x[0]&&y[1]+h[1]<y[0]+h[0])
		{
			answer=w[1]*h[1]+w[0]*h[0]-2*h[1]*(x[1]+w[1]-x[0]);
		}
		else if(x[1]+w[1]<x[0]+w[0]&&x[1]+w[1]>x[0]&&y[1]+h[1]==y[0]+h[0])
		{
			answer=w[1]*h[1]+w[0]*h[0]-2*h[1]*(x[1]+w[1]-x[0]);
		}
		
		else if(x[1]+w[1]>x[0]+w[0]&&y[0]+h[0]<y[1]+h[1]&&y[0]+h[0]>y[1])
		{
			answer=w[1]*h[1]+w[0]*h[0]-2*w[0]*(y[0]+h[0]-y[1]);
		}
		else if(x[1]+w[1]==x[0]+w[0]&&y[0]+h[0]<y[1]+h[1]&&y[0]+h[0]>y[1])
		{
			answer=w[1]*h[1]+w[0]*h[0]-2*w[0]*(y[0]+h[0]-y[1]);
		}
		
		else if(x[1]+w[1]<x[0]+w[0]&&x[1]+w[1]>x[0]&&y[0]+h[0]<y[1]+h[1]&&y[0]+h[0]>y[1])
		{
			answer=w[1]*h[1]+w[0]*h[0]-2*(x[1]+w[1]-x[0])*(y[0]+h[0]-y[1]);
		}
		
		else
		{
			answer=w[1]*h[1]+w[0]*h[0];
		}		
	}
	
	if((x[0]<=x[1])&&(y[0]>y[1]))
	{
		if(x[0]+w[0]>x[1]+w[1]&&y[0]+h[0]<y[1]+h[1])
		{
			answer=w[0]*h[0]+w[1]*h[1]-2*w[1]*h[0];
		}
		else if(x[0]+w[0]==x[1]+w[1]&&y[0]+h[0]<y[1]+h[1])
		{
			answer=w[0]*h[0]+w[1]*h[1]-2*w[1]*h[0];
		}
		else if(x[0]+w[0]>x[1]+w[1]&&y[0]+h[0]==y[1]+h[1])
		{
			answer=w[0]*h[0]+w[1]*h[1]-2*w[1]*h[0];
		}
		else if(x[0]+w[0]==x[1]+w[1]&&y[0]+h[0]==y[1]+h[1])
		{
			answer=w[0]*h[0]+w[1]*h[1]-2*w[1]*h[0];
		}

		else if(x[0]+w[0]<x[1]+w[1]&&x[0]+w[0]>x[1]&&y[0]+h[0]<y[1]+h[1])
		{
			answer=w[1]*h[1]+w[0]*h[0]-2*h[0]*(x[0]+w[0]-x[1]);
		}
		else if(x[0]+w[0]<x[1]+w[1]&&x[0]+w[0]>x[1]&&y[0]+h[0]==y[1]+h[1])
		{
			answer=w[1]*h[1]+w[0]*h[0]-2*h[0]*(x[0]+w[0]-x[1]);
		}
		
		else if(x[0]+w[0]>x[1]+w[1]&&y[1]+h[1]<y[0]+h[0]&&y[1]+h[1]>y[0])
		{
			answer=w[1]*h[1]+w[0]*h[0]-2*w[1]*(y[1]+h[1]-y[0]);
		}
		else if(x[0]+w[0]==x[1]+w[1]&&y[1]+h[1]<y[0]+h[0]&&y[1]+h[1]>y[0])
		{
			answer=w[1]*h[1]+w[0]*h[0]-2*w[1]*(y[1]+h[1]-y[0]);
		}
		
		else if(x[0]+w[0]<x[1]+w[1]&&x[0]+w[0]>x[1]&&y[1]+h[1]<y[0]+h[0]&&y[1]+h[1]>y[0])
		{
			answer=w[1]*h[1]+w[0]*h[0]-2*(x[0]+w[0]-x[1])*(y[1]+h[1]-y[0]);
		}
		
		else
		{
			answer=w[1]*h[1]+w[0]*h[0];
		}		
	}
	
		
	printf("%d\n",answer);
}
