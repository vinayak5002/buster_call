#include <bits/stdc++.h>
using namespace std;

int main(){
    vector<vector<int>>ã€€nx(1,vector<int>(2,0));
    vector<int> temp(2);
  
    for(;;){
        cin >> temp[0] >> temp[1];
        
        if(temp[0] == 0 && temp[1] == 0) break;
        nx.push_back(temp);
        
    }

    int count;
    vector<int> ans(nx.size()-1,0);
    for(int l = 1; l < nx.size()-1; l++){
        count = 0;
        for(int i = 1; i <= nx[l][0]-2; i++){
            for(int j = i+1; j <= nx[l][0]-1; j++){
                for(int k = j+1; k <= nx[l][0]; k++){
                    if(i+j+k == nx[l][1]) count++;
                }
            }
        }
        ans[l] = count;
    }
    
    for(int i = 0; i < ans.size(); i++){
        cout << ans[i] << endl;
    }
}
