int main(void) {
    sonic();
    int n;
    cin >> n;
    vector<string> s(n);
    cin >> s;

    rep(_, 2) {
        int mi = 0, sum = 0;
        vector<pair<int, int>> v;
        rep(i, n) {
            int x = 0, y = 0;
            for (auto c : s[i]) {
                if (c == '(')
                    ++x;
                else
                    --x;
                chmin(y, x);
            }
            if (x == 0)
                chmin(mi, y);
            else if (x > 0) {
                v.emplace_back(y, x);
            }
            sum += x;
        }

        if (sum) {
            No();
            return 0;
        }

        sort(all(v));
        reverse(all(v));

        sum = 0;
        for (auto p : v) {
            if (p.first + sum < 0) {
                No();
                return 0;
            }
            sum += p.second;
        }

        if (sum + mi < 0) {
            No();
            return 0;
        }

        rep(i, n) {
            for (auto &c : s[i]) {
                if (c == '(')
                    c = ')';
                else
                    c = '(';
            }
            reverse(all(s[i]));
        }
    }
    Yes();

    return 0;
}
