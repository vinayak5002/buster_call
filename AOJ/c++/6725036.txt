#include<bits/stdc++.h>
using namespace std;
typedef long long int ll;
#define rep(i, n) for(ll i = 0; i < (ll)n; i++)
#define ALL(v) (v).begin(), (v).end()

ll GCD(ll a, ll b) {
  return b ? GCD(b, a % b) : a;
}




vector<ll> divisor(ll n) {
  // O(âˆšn)
  vector<ll> ans;
  for (ll i = 1; i * i <= n; i++) {
    if (n % i == 0) {
      ans.push_back(i);
      if (i * i != n) ans.push_back(n / i);
    }
  }
  sort(ALL(ans));
  return ans;
}

}
