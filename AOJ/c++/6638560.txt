void solve(int n) {
  vector< int > rs(n);
  whole(iota, rs, 0);
  range(i, 1, rs.size()) rs[i] += rs[i - 1];

  int ans = 0;
  range(r, 1, rs.size()) range(l, 0, r) {
    if (rs[r] - rs[l] == n) ans++;
  }

  cout << ans << endl;
}

signed main() {
  int n;

  while (cin >> n, n) {
    solve(n);
  }
}
