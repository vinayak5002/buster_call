#include<stdio.h>
#include<ctype.h>
int main(){
    int i,j,cnt;
    i=0;
    char alph[26] = {'a','b','c','d','e','f','g','h','i','j','k','l','m','n','o','p','q','r','s','t','u','v','w','x','y','z'};
    char str[1200]={0};
    char CNT[26]={0};
    scanf("%s",&str);
    for(j=0;j<26;j++){
        for(i=0;i<(str[i]!='\0');i++){
            str[i]=tolower(str[i]);
            if(str[i]==alph[j])cnt++;
        }
        CNT[j]=cnt;
        cnt = 0;
    }
    for(i=0;i<26;i++){
        printf("%sï¼š%s\n",alph[i],CNT[i]);
    }
    return 0;
}
