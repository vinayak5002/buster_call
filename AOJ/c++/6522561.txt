#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int main()
{
    ll a, b;
    cin >> a >> b;
    if (a > b)
    {
        cout << "a > b"
             << "\n";
    }
    else if (a < b)
    {
        cout << "a < b"
             << "\n";
    }
    else
    {
        cout << "a == b"
             << "\n";
    }
}
