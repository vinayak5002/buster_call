#include<stdio.h>
int main(){
	int a[100][100],b[100][100],i,j,k,n,m,l,x;
	long long int c[100][100]={0};
	scanf("%d %d %d",&n,&m,&l);
    printf("\n");
	for(i=0;i<n;i++){
		for(j=0;j<m;j++){
			scanf("%d",&x);
			a[i][j]=x;
		}
	}

	for(j=0;j<m;j++){
		for(k=0;k<l;k++){
			scanf("%d",&x);
			b[j][k]=x;
		}
	}

	for(i=0;i<n;i++){
		for(k=0;k<l;k++){
			for(j=0;j<m;j++){
				c[i][k]+=a[i][j]*b[j][k];
			}
		}
	}

	for(i=0;i<n;i++){
		for(k=0;k<l;k++){
			printf("%lld ",c[i][k]);
		}
		printf("\n");
	}
	return 0;
}
