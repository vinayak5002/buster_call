#include <bits/stdc++.h>
using namespace std;

int main()
{
    int m, f, r;
    while (true)
    {
        cin >> m >> f >> r;
        if (m == f && m == r && m == -1)
        {
            break;
        }
        else if ((m + f) >= 80)
            cout << "A" << endl;
        else if ((m + f) >= 65)
            cout << "B" << endl;
        else if ((m + f) >= 50 || r >= 50)
            cout << "C" << endl;
        else if ((m + f) >= 30)
            cout << "D" << endl;
        else
            cout << "F" << endl;
    }
}
