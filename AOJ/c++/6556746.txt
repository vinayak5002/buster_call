n = int(input())
A = list(map(int, input().split()))
q = int(input())
m = list(map(int, input().split()))

candidate = set()

for i in range(2**n):
    num = 0
    for j in range(n):
        if ((i >> j) & 1):
            num += A[j]
    candidate.add(num)
            
for i in range(q):
    if m[i] in candidate:
        print("yes")
    else:
        print("no")
