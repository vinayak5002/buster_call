#include <iostream>
using namespace std;

int main()
{
	const int s[24][3] = { {1, 2, 3}, {1, 3, 5}, {1, 5, 4}, {1, 4, 2},
				{2, 1, 4}, {2, 4, 6}, {2, 6, 3}, {2, 3, 1},
				{3, 1, 2}, {3, 2, 6}, {3, 6, 5}, {3, 5, 1},
				{4, 1, 5}, {4, 5, 6}, {4, 6, 2}, {4, 2, 1},
				{5, 1, 3}, {5, 3, 6}, {5, 6, 4}, {5, 4, 1},
				{6, 2, 4}, {6, 4, 5}, {6, 5, 3}, {6, 3, 2} };
	
	int a[6] = { 0 }, b[6] = { 0 }, n = 0;

	for (int i = 0; i < 6; i++)
		cin >> a[i];

	for (int i = 0; i < 6; i++)
		cin >> b[i];

	for (int i = 0; i < 24; i++) {
		if (a[0] == b[s[i][0] - 1]) {
			if (a[1] == b[s[i][1] - 1]) {
				if (a[2] == b[s[i][2] - 1]) {
					if (a[3] == b[6 - s[i][2]]) {
						if (a[4] == b[6 - s[i][1]]) {
							if (a[5] == b[6 - s[i][0]]) {
								n = 1;
							}
						}
					}
				}
			}
		}
	}

	if(n == 1)
		cout << "Yes" << endl;
	else
		cout << "No" << endl;

	return 0;
}
