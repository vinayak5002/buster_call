#include <stdio.h>
#include <stdlib.h>

int main()
{
    int grid[256][256], n, tempA=0, tempB=0, max=0, con[100000]={0}, c, i, j, k;
    char bin;
    for(c=1;1;c++)
    {
        scanf("%d",&n);
        if(n==0)
            break;
        for(i=1;i<=n;i++)
        {
            scanf("%c",&bin);
            for(j=n;j>0;j--)
            {
                grid[j][i]=bin%10;
                bin=bin/10;
            }
        }
        for(i=1;i<=n;i++)
        {
            for(j=1;j<=n;j++)
            {
                if(grid[j][i]==1)
                {
                    tempA++;
                    max=(tempA>max)?tempA:max;
                }
                else
                {
                    tempA=0;
                }
                if(grid[i][j]==1)
                {
                    tempB++;
                    max=(tempB>max)?tempB:max;
                }
                else
                {
                    tempB=0;
                }
            }
            tempA=tempB=0;
        }
        for(k=n-1;k>0;k--)
        {
            for(i=k,j=1;i<=n;i++,j++)
            {
                if(grid[j][i]==1)
                {
                    tempA++;
                    max=(tempA>max)?tempA:max;
                }
                else
                {
                    tempA=0;
                }
                if(grid[i][j]==1)
                {
                    tempB++;
                    max=(tempB>max)?tempB:max;
                }
                else
                {
                    tempB=0;
                }
            }
            tempA=tempB=0;
        }
        for(k=n-1;k>0;k--)
        {
            for(i=k,j=n;i<=n;i++,j--)
            {
                if(grid[j][i]==1)
                {
                    tempA++;
                    max=(tempA>max)?tempA:max;
                }
                else
                {
                    tempA=0;
                }
            }
            tempA=0;
        }
        for(k=2;k<n;k++)
        {
            for(i=1,j=k;j>0;i++,j--)
            {
                if(grid[j][i]==1)
                {
                    tempA++;
                    max=(tempA>max)?tempA:max;
                }
                else
                {
                    tempA=0;
                }
            }
            tempA=0;
        }
        con[c]=max;
        max=0;
    }
    for(i=1;i<c;i++)
    {
        printf("%d\n",con[i]);
    }
    return 0;
}
