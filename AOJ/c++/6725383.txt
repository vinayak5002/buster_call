void solve(){
	int N;
	cin>>N;
	vector<pair<int,int>> p(N);
	rep(i,N) cin>>p[i].first>>p[i].second;
	So(p);
	set<int> s;
	rep(i,N){
		p[i].first--;
		p[i].second--;
		s.insert(p[i].first);
		s.insert(p[i].second+N);
	}
	if((int)s.size()!=N*2){
		cout<<"-1\n";
		return ;
	}
	vector<int> seen(N);
	int ans=N;
	rep(i,N){
		if(seen[i]) continue;
		ans--;
		int ind=i;
		while(seen[ind]==0){
			seen[ind]=1;
			ind=p[ind].second;
		}
	}
	cout<<ans<<"\n";
}
