#include <iostream>
#include <iomanip>
#include <cmath>
#include <climits>
#include <string.h>
using namespace std;
int main()
{
	int n
	cin >> n;	
	int s1 = 0, s2 = 0;
	while (n--)
	{
		string s, t;
		cin >> s >> t;
		if (s>t){
			s1 +=3
		}
		else if (s < t) {
			s2 += 3
		}
		else {
			s1++;
			s2++;
		}
		cout << s1 << s2 << endl;
	}
}
