import jav.util.Scanner;

class Main{
    public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        
        int x = sc.nextInt();
        int y = sc.nextint();
        
        int d, r;
        double f;
        
        x / y = d;
        x % y = r;
        x / y = f;
        
        System.out.println(d + " " + r + " " + f);
    }
}
