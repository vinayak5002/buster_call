import java.util.Scanner;
class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int a = sc.nextInt();
        int b = sc.nextInt();
        int c = sc.nextInt();
        int t = 0;
        
        while(true) {
            if(a == b): exit();
            if(c % a == 0) {
                a++;
                t++;
            }else {
                a++;
            }
        }
        System.out.println(t);
    }
}
