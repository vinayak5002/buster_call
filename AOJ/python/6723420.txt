import math

# Triangle a, b = "side". C = "degrees"ã€€
a, b, C = map(float, input().split())
C = C * math.pi / 180.0
# Triangle Area = "S"
S = 0.5 * a * b * math.sin(C)

# Triangle circumference = "L"
c = math.sqrt(a * a + b * b - 2 * a * b *math.cos(C))
L = a + b + c

# Triangle height = "h"

h = 2 * S / a

# output
print(S)
print(L)
print(h)
