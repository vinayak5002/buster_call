/////////////////////////////////////////////
// プログラム名 : ITP1_4_D(Max,Min,Sum)
// 内容        : 入力したn個の整数の最大値、最小値、合計を算出する
/////////////////////////////////////////////
#include <stdio.h>

/**********************************************************/
/* 機能　 > 入力したn個の整数の最大値、最小値、合計を算出する  */
/* 入力値 > なし                                           */
/* 出力値 > 0：正常処理　                                   */
/**********************************************************/
int main(void) {

    int      i;
    int      n;                                 // 入力する整数の個数
    int      num[1000];                         // 入力する整数の値
    int      max = -1000000;                    // 最大値
    int      min = 1000000;                     // 最小値
    long int sum = 0;                           // 合計 

    scanf("%d", &n);
    
    for(i = 0; i < n; i++) {
        scanf("%d", &num[i]);

        if(num[i] > max) {                      // 最大値の算出
            max = num[i];
        }

        if(num[i] < min) {                      // 最小値の算出
            min = num[i];
        }

        sum += num[i];                          // 合計の算出        
    }

    printf("Min Max Sum : %d %d %d\n", min, max, sum);

    return 0;    
}
