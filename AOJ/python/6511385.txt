import java.util.*;

class Main{
    public static void main(String[] args){
        Scanner scan = new Scanner(System.in);
        int a = scan.nextInt();
        int b = scan.nextInt();
        d = b / a;
        r = b % a;
        f = (double)b / a;
        System.out.print(d+" "+r+" "+f);
    }
}
