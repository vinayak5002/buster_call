/////////////////////////////////////////////
// プログラム名 : ITP1_5_A_Rectangle
// 内容        : H×Wの長方形を#で描く
/////////////////////////////////////////////
#include <stdio.h>

#define N   1000                                    // データセットの個数
#define MAX 300                                     // 入力値の最大値

/**********************************************************/
/* 機能　 > 入力したn個の整数の最大値、最小値、合計を算出する  */
/* 入力値 > なし                                           */
/* 出力値 > 0：正常処理　                                   */
/**********************************************************/
int main(void) {

    unsigned int H[N];                              // 長方形の縦[cm]
    unsigned int W[N];                              // 長方形の横[cm]
    int          k;
    int          i;
    int          j;
    int          cnt = 0;                           // カウントを0に初期化

    for(k = 0; k < N; k++) {
        scanf("%d %d", &H[k], &W[k]);

                if((H[k] == 0) && (W[k] == 0)) {
                    break;                          // 入力が0のときループを抜ける
                }

                if((H[k] > MAX) || (W[k] >MAX)) {
                    break;
                }

                cnt++;                              // 入力値が0、または300を超えるまでの回数をカウント
    }                                              


    for(k = 0; k < cnt; k++) {                      // データセットごとに長方形の表示
        for(j = 0; j < H[k]; j++) {                 // 縦の表示
            for(i = 0; i < W[k]; i++) {             // 横の表示
                printf("#");
            }
            printf("\n"); 
        }
        printf("\n");
    }

    return 0;
}
