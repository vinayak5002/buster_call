import java.util.Scanner;
public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        while(true){
            String str=sc.nextLine();
            if (str.equals("-")){
                break;
            }
            int m=sc.nextInt();
            for(int i=0;i<m;i++){
                int hi=sc.nextInt();
                str=str.substring(hi)+str.substring(0,hi);
            }
            System.out.println(str); 
        }
        sc.close();
    }
}
