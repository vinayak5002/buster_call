/////////////////////////////////////////////
// プログラム名 : ITP1_5_A_Rectangle
// 内容        : 長方形を#で描く
/////////////////////////////////////////////
#include <stdio.h>

#define NUMBER 1000                                 // データセットの個数の最大値
#define MAX    300                                  // 入力値の最大値
#define MIN    3                                    // 入力値の最小値

/********************************************/
/* 機能　 > 長方形を#で描く                   */
/* 入力値 > なし                             */
/* 出力値 > 0：正常処理　                     */
/********************************************/
int main(void) {

    unsigned int H[NUMBER];                         // 長方形の縦[cm]
    unsigned int W[NUMBER];                         // 長方形の横[cm]
    int          k;
    int          i;
    int          j;
    int          cnt = 0;                           // カウントを0に初期化

    for(k = 0; k < NUMBER; k++) {                   // データセットの入力
        scanf("%d %d", &H[k], &W[k]);

        if((H[k] == 0) && (W[k] == 0)) {            // 入力が0のとき
            break;
        }
        
        if((H[k] > MAX) || (W[k] >MAX) || (H[k] < MIN) || (W[k] < MIN)) {
                                                    // 入力が300より大きい、3より小さいとき
            break;
        }

        cnt++;                                      // 入力値が0、または300を超えるまでの回数をカウント
    }                                              


    for(k = 0; k < cnt; k++) {                      // データセットごとに長方形の表示
        for(j = 0; j < H[k]; j++) {                 // 縦の表示
            if((j == 0) || (j == H[k] - 1)) {       // 横（上辺と底辺）の表示
                for(i = 0; i < W[k]; i++) {         // 横の表示
                    printf("#");
                }
                printf("\n");
            }
            else {                                  // 横（上辺と底辺以外）の表示
                for(i = 0; i < W[k]; i++) {
                    if((i == 0) || (i == W[k] - 1)) {
                                                    // 縦（左辺と右辺）の表示
                        printf("#");
                    }
                    else {                          // 縦（左辺と右辺以外）の表示
                        printf(".");
                    }
                }
                printf("\n");
            }
        }
        printf("\n");                               // 各データセットの後に空行
    }

    return 0;
}
