import java.util.Scanner;

public class Main{
public static void main(String[] args){
int a,b;

Scanner sc=new Scanner(System.in);
a=sc.nextInt();
b=sc.nextInt();

if(a==b){
System.out.println(a+"=="+b);
}
else if(a<b){
System.out.println(a+"<"+b);
}
else if{
System.out.println(a+">"+b);
}
sc.close();
}
}
