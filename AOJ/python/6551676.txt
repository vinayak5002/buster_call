import java.util.Scanner;

class Main{
    public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        
        for(int i=1;i<10000;i++){
            int x = sc.nextInt();
            System.out.println("Case "+i+": "+x);
            if(x==0){
                break;
            }
        }
    }
}
