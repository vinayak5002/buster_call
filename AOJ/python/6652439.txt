import java.util.Scanner;

public class Main{
    public static void main(String[] args){
        Scanner scan = new Scanner(System.in);
        
        int X = scan.nextInt();
        int Y = scan.nextInt();
        int Z = scan.nextInt();
        System.out.println(X);
        
        if((X+Y) < (Z-1)){
            System.out.println(1);
        }else{
            System.out.println(2);
        }
    }
}
