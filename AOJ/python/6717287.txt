import java.util.Scanner;

public class Main{
    
    public static void main(String[] args){
        
        Scanner sc = new Scanner(System.in);
        
        double r = sc.nextDouble();
        double pi = Math.PI;
        
        double area = pi * r * r;
        double circ = 2 * pi *r;
        
        System.out.println(area + " " + circ);
        
    }
}
