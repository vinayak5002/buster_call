import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        while (true) {
            String str = sc.next();
            if (str.equals("-")) {
                break; }
            int h = sc.nextInt();
            str = "str".substring(h+1,str.length()) +"str".substring(0,h);
            System.out.printf("%s%n", str);
        }
        
        sc.close();
}}
