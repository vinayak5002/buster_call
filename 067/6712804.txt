#inputする
N = int(input())
A = list(map(int,input().split()))
#例えばA =[5,4,3,2,1]
#A[0]は整頓済みとする。よって、A[1]~A[N-1]までをループする
for i in range(1,N):
    v = A[i]
    #vに4を保存する
    j = i-1
    #iより左の数、つまり「整頓済み」の数字を見ていく
    while j>= 0 and v < A[j]:
        #jを徐々に-していき、0になるまで、
        #つまりは整頓済みのものを全てチェックしていく
        #そして、v < A[j]、保存した数よりも大きい数字は1つ右にずれていく
        A[j+1] = A[j]
        j -= 1
    A[j+1] = v
    print(" ".join(map(str,A)))
    #この時、Jは最大でも0である（）j=-1の時点でループ終了A[0]=vとなる
    #最後に大小の境界に保存した数字を代入する
