import java.util.Scanner;

class Main{
    public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        int a = sc.nextInt();
        int b = sc.nextInt();
        int c = sc.nextInt();
        
        int i=0;
        int j=0;
        
        for(i=a ; i>b ; i++){
            if(c%i == 0) j++;
        }
        
        System.out.println(j);
    }
}
