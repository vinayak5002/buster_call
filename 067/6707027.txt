#include<bits/stdc++.h>
using namespace std;

bool isprime(int x){
    if(x==2) return true;

    if(x<2 || x%2==0) return false;

    int i = 3;
    while(i<=sqrt(x)){
        if(x%i==0) return false;
        i = i+2;
    }

    return true;
}

int main(){
    int n, x, cnt=0;
    cin >> n;
    for(int i=0; i<n; i++){
        cin >> x;
        if(isprime(x)) cnt++;
    }

    cout << cnt << endl;
    
    return 0;
}
