import java.util.Scanner;

class Main{
    public static void main(String[] arg){
        Scanner sc = new Scanner(System.in);
        int x = sc.nextInt();
        int y = sc.nextInt();
     while(true){
        if(x != 0 && y != 0);
         if(x < y){
             system.out.println(x +" "+ y);
         }else{
             system.out.println(y +" "+ x);
             
         }
     }
        
    }
}
