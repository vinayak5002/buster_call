const int MOD = 1000000007;
/*
  冪乗計算。
  nを2進数変換して計算すれば、計算回数をO(log(n))にできる。
*/
int main() {
    ll m, n;
    cin >> m >> n;
    ll res = 1;

    while (n > 0) {
        if (n & 1) res = res * m % MOD;
        m = m * m % MOD;
        n >>= 1;
    }
    cout << res << endl;
}
