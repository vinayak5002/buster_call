import sys
input = sys.stdin.readline


def print2D(matrix):
    print("\n".join(" ".join(map(str, v)) for v in matrix))


res = []
while 1:
    b = int(input())
    if b == 0:
        break
    for i in range(4 * 10 ** 5 - 1, 0, -1):
        k = 2 * b - i * i + i
        if k > 0 and k % (2 * i) == 0:
            res.append((k // (2 * i), i))
            break

print2D(res)
