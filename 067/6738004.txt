#include <iostream>
using namespace std;

int main()
{
    int a;
    cin >> a;
    int b;
    cin >> b;
    int c;
    cin >> c;
    if (a <= b && a <= c) {
        cout << a << min(b, c) << max(b, c);
    }
    else if (b <= a && b <= c) {
        cout << b  << min(a, c) << max(a, c);
    }
    else if (c <= b && c <= a) {
        cout << c << min(a, b) << max(a, b);
    }
}
