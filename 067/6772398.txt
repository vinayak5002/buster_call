#include<stdio.h>

int main(void){
 
 int a,b;
 char op;

for(;;){
     scanf("%d %c %d",&a,&op,&b);

      if(op == '+'){
         printf("%d\n",a + b);
     }else if(op == '-'){
         printf("%d\n",a - b);
     }else if(op=='*'){
         printf("%d\n",a * b);
     }else if(op=='/'){
         printf("%d\n",a/b);
     }else if(op=='?'){
         break;
     }
}

return 0;

}
