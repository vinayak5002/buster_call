bool solve(){
    int N;
    cin >> N;
    if(N == 0) return false;
    vector<int> A(N);
    for(int i=0; i<N; i++){
        cin >> A[i];
    }
    sort(A.begin(),A.end());
    int ans = 1e9;
    for(int i=0; i<N-1; i++){
        chmin(ans,A[i+1]-A[i]);
    }
    cout << ans << endl;
    return true;
}


int main(){
    while(solve());
    return 0;
