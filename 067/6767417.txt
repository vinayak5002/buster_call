#include <stdio.h>

int main(void)
{
  int n, i, v, j, k;

  printf("n = ");
  scanf("%d", &n);

  int a[n];

  for (i = 0; i < n; i++)
  {
    scanf("%d", &a[i]);
  }

  // for (i = 0; i < n; i++)
  // {
  //   printf("%d ", a[i]);
  // }
  // printf("\n");

  for (i = 1; i < n; i++)
  {
    v = a[i];
    j = i - 1;
    while (j >= 0 && a[j] > v)
    {
      a[j + 1] = a[j];
      j--;
    }
    // printf("j=%d\n", j);
    a[j + 1] = v;
    for (k = 0; k < n; k++)
    {
      printf("%d ", a[k]);
    }
    printf("\n");
  }

  // printf("%d\n", n);
  return 0;
}
