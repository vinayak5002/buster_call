#include<stdio.h>
int main(void){
    int n,a[1000],c=0,d=0,i,j,m=0,o=0,s=0;
    scanf("%d",&n);
    for(i=0;i<n;i++){
        scanf("%d",&a[i]);
        if(i>0){
            m=a[i-1];
            o=a[i-1];
        }
        for(j=0;j<5;j++){
            if(a[i]==m){
                break;
            }
            m++;
            if(m==10){
                m=0;
            }
            c++;
        }
        for(j=0;j<5;j++){
            if(a[i]==o){
                break;
            }
            o--;
            if(o==-1){
                o==9;
            }
            d++;
        }
        if(c<d){
            s+=c;
        }else if(c>d){
            s+=d;
        }else{
            s+=5;
        }
    }
    printf("%d\n",s);
}
