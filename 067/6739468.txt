"""
A:nÃ—m
b:m    
"""


n,m=map(int,input().split())

A=[list(map(int,input().split()))for i in range(n)]
b=[int(0)for j in range(m)]

for j in range(m):
    b[j]=int(input())

c=[int(0)for i in range(n)]

for i in range(n):
    for j in range(m):
        c[i]+=A[i][j]*b[j]
        
for i in range(n):
    print(c[i])
