#include <bits/stdc++.h>
using namespace std;

int main(){
    int n;
    cin >> n;
    vector<int> a(n);
    for(int i = 0; i < n; ++i) cin >> a[i];
    
    sort(a.begin(), a.end());
    
    long long sum = accumulate(a.begin(), a.end(), 0);
    
    cout << a[0] << " " << a[n - 1] << " "
    << sum << endl;
    
    return 0;
}
