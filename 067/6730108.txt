struct Monoid {
    using value_type = pair<int, int>;
    static constexpr value_type id = {0, 0};
    static constexpr value_type op(const value_type& lhs, const value_type& rhs) {
        return {lhs.first + rhs.first, std::max(lhs.second, rhs.second)};
    }

    static constexpr value_type f(const value_type& v) {
        return {v.first + 1, v.second + 1};
    }
};

template <class M>
struct ReRooting {
  private:
    using T = typename M::value_type;

  public:
    ReRooting(const Graph<void>& g) : graph(g), dp(g.size()), data(g.size()) {
        this->build();
    }

    const T& operator[](int i) const {
        return data[i];
    }
    T& operator[](int i) {
        return data[i];
    }

  private:
    Graph<void> graph;
    std::vector<std::vector<T>> dp;
    std::vector<T> data;

    void build() {
        this->dfs(0);
        this->bfs(0);
    }

    T dfs(int v, int p = -1) {
        T res = M::id;
        int deg = graph[v].size();
        dp[v] = std::vector<T>(deg, M::id);
        for (int i = 0; i < deg; ++i) {
            int u = graph[v][i].to();
            if (u == p)
                continue;
            dp[v][i] = dfs(u, v);
            res = M::op(res, dp[v][i]);
        }
        return M::f(res);
    }
    void bfs(int v, int p = -1, T dp_p = M::id) {
        int deg = graph[v].size();
        for (int i = 0; i < deg; i++) {
            if (graph[v][i].to() == p)
                dp[v][i] = dp_p;
        }
        std::vector<T> dp_l(deg + 1, M::id), dp_r(deg + 1, M::id);
        for (int i = 0; i < deg; ++i) {
            dp_l[i + 1] = M::op(dp_l[i], dp[v][i]);
        }
        for (int i = deg - 1; i >= 0; --i) {
            dp_r[i] = M::op(dp_r[i + 1], dp[v][i]);
        }
        data[v] = M::f(dp_l[deg]);
        for (int i = 0; i < deg; ++i) {
            int u = graph[v][i].to();
            if (u == p)
                continue;
            this->bfs(u, v, M::f(M::op(dp_l[i], dp_r[i + 1])));
        }
    }
};
