template <class T>
inline bool chmin(T &a, T b) {
    if (a > b) {
        a = b;
        return true;
    }
    return false;
}

template <class T>
inline bool chmax(T &a, T b) {
    if (a < b) {
        a = b;
        return true;
    }
    return false;
}

template <class T>
T pow(T x, i64 n) {
    T res = 1;
    while (n > 0) {
        if (n & 1) res = res * x;
        x = x * x;
        n >>= 1;
    }
    return res;
}

template <class T>
T mod_pow(T x, i64 n, i64 mod) {
    T res = 1;
    while (n > 0) {
        if (n & 1) res = (res * x) % mod;
        x = (x * x) % mod;
        n >>= 1;
    }
    return res;
}

    }

constexpr i64 LNF = std::numeric_limits<i64>::max() / 4;

constexpr int INF = std::numeric_limits<int>::max() / 2;

const std::vector<int> dy = {1, 0, -1, 0, 1, 1, -1, -1};
const std::vector<int> dx = {0, 1, 0, -1, 1, -1, 1, -1};

}  // namespace ebi

namespace ebi {

void main_() {
    int h,w,n;
    std::cin >> h >> w >> n;
    std::vector<std::string> s(h);
    std::cin >> s;
    std::vector<std::pair<int,int>> points(n);
    for(auto &[y, x]: points) {
        std::cin >> y >> x;
        y--; x--;
    }
    std::vector dp(h, std::vector<int>(w, -1));
    auto bfs = [&](int sy, int sx) -> void {
        std::deque<std::pair<int,int>> que;
        std::vector seen(h, std::vector<int>(w, INF));
        seen[sy][sx] = 0;
        que.push_back({sy, sx});
        while(!que.empty()) {
            auto [y, x] = que.front();
            que.pop_front();
            rep(i,0,4) {
                int ny = y + dy[i];
                int nx = x + dx[i];
                if(0 <= ny && ny < h && 0 <= nx && nx < w && s[ny][nx] != '#' && seen[ny][nx] == INF) {
                    if(s[ny][nx] == '.') {
                        seen[ny][nx] = seen[y][x];
                        que.push_front({ny, nx});
                    }
                    else {
                        assert(s[ny][nx] == '@');
                        seen[ny][nx] = seen[y][x] + 1;
                        que.push_back({ny, nx});
                    }
                }
            }
        }
        rep(i,0,h) rep(j,0,w) {
            chmax(dp[i][j], seen[i][j]);
        }
    };
    for(auto [y, x]: points) {
        bfs(y, x);
    }
    int y = -1, x = -1;
    int ret = INF;
    rep(i,0,h) rep(j,0,w) {
        if(dp[i][j] < ret) {
            y = i+1;
            x = j+1;
            ret = dp[i][j];
        }
    }
    std::cout << y << " " << x << '\n';
}

}  // namespace ebi

int main() {
    std::cout << std::fixed << std::setprecision(15);
    std::cin.tie(nullptr);
    std::ios::sync_with_stdio(false);
    int t = 1;
    // std::cin >> t;
    while (t--) {
        ebi::main_();
    }
    return 0;
}
