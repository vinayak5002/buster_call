#include <iostream>
#include <vector>
using namespace std;
int main()
{
    int n;
    cin >> n;
    int d = 0;
    vector<int> v(n);
    for (int i = 0;i < n;i++) {
        cin >> d;
        v.push_back(d);
    }
    reverse(v.begin(), v.end());
    for (int i = 0;i < n;i++) {
        cout << v[i]<<" ";
    }
}
