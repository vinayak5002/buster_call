while True:
    a = int(input("enter first number"))
    op = input("please enter the operator")
    b = int(input("Please enter the second number"))
    if op == "+" :
        print(a+b)
    if op == "-":
        print(a-b)
    if op == "*":
        print(a*b)
    if op =="/":
        print(a/b)
    if op =="?":
        break
