def cmb(n,r):
    r = min(n-r,r)
    if r == 0: return 1
    over = reduce(mul, range(n, n - r, -1))
    under = reduce(mul, range(1,r + 1))
    return over // under
def combnations_count(n, r):
    return math.factorial(n) // (math.factorial(n-r) * math.factorial(r))
