import sys

readline = sys.stdin.readline

def main():
    N, W = map(int, readline().split())
    A = [list(map(int, readline().split())) for _ in range(N)]
    dp = [0] * (W+1)

    for i in range(N):
        v, w = A[i]
        for j in range(W+1):
            if j - w >= 0:
                dp[j] = max(dp[j], dp[j-w] + v)

    print(dp[-1])

if __name__ == '__main__':
    main()
